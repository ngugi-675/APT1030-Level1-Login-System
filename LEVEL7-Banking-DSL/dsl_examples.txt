# Banking DSL Example Commands
# ============================================

# Account Creation
CREATE ACCOUNT SAVINGS WITH 10000
CREATE ACCOUNT CHECKING WITH 5000
CREATE ACCOUNT INVESTMENT
CREATE ACCOUNT EMERGENCY WITH 20000

# Deposits
DEPOSIT 5000 TO SAVINGS
DEPOSIT 10000 TO INVESTMENT
DEPOSIT 2500 TO CHECKING

# Withdrawals
WITHDRAW 1000 FROM CHECKING
WITHDRAW 2000 FROM SAVINGS
WITHDRAW 500 FROM EMERGENCY

# Balance Checks
BALANCE SAVINGS
BALANCE CHECKING
BALANCE INVESTMENT
BALANCE EMERGENCY

# Transfers (Simple)
TRANSFER 2000 FROM SAVINGS TO CHECKING
TRANSFER 5000 FROM INVESTMENT TO SAVINGS

# Transfers with Conditions
TRANSFER 3000 FROM SAVINGS TO INVESTMENT IF BALANCE > 5000
TRANSFER 15000 FROM EMERGENCY TO SAVINGS IF BALANCE > 20000
TRANSFER 1000 FROM CHECKING TO SAVINGS IF BALANCE > 2000

# Transaction History
HISTORY SAVINGS
HISTORY CHECKING
HISTORY INVESTMENT

# Invalid Commands (For Testing)
TRANSFER 5000 FROM X TO Y              # Non-existent accounts
TRANSFER -100 FROM A TO B              # Negative amount
WITHDRAW 999999 FROM SAVINGS           # Insufficient funds
TRANSFER 5000 FROM SAVINGS TO CHECKING IF BALANCE > 100000  # Condition fails
DEPOSIT -500 TO SAVINGS                # Negative deposit

# Complex Scenario: Salary Deposit and Bill Payments
CREATE ACCOUNT SALARY WITH 50000
CREATE ACCOUNT RENT
CREATE ACCOUNT UTILITIES
CREATE ACCOUNT GROCERIES

DEPOSIT 50000 TO SALARY
TRANSFER 15000 FROM SALARY TO RENT IF BALANCE > 20000
TRANSFER 5000 FROM SALARY TO UTILITIES IF BALANCE > 10000
TRANSFER 10000 FROM SALARY TO GROCERIES IF BALANCE > 5000
BALANCE SALARY
